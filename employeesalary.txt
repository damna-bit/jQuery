public class Employee {

    private String name;
    private int employeeID;
    private double salary;

    public Employee(String name, int employeeID, double salary) {
        this.name = name;
        this.employeeID = employeeID;
        this.salary = salary;
    }

   
    public double calculateYearlySalary() {
        return salary * 12; 
    }


    public void giveRaise(double raisePercentage) {
        if (raisePercentage > 0) {
            double raiseAmount = salary * raisePercentage / 100;
            salary += raiseAmount;
            System.out.println("Raise of " + raisePercentage + "% given. New salary: " + salary);
        } else {
            System.out.println("Invalid raise percentage.");
        }
    }

  
    public void displayInfo() {
        System.out.println("Employee Information:");
        System.out.println("Name: " + name);
        System.out.println("Employee ID: " + employeeID);
        System.out.println("Salary: " + salary);
    }

    
    public static void main(String[] args) {
        Employee emp = new Employee("Arun", 003, 17000.0);
        emp.displayInfo();
        System.out.println("Yearly Salary: " + emp.calculateYearlySalary());
        emp.giveRaise(5); 
        emp.displayInfo();
    }
}
